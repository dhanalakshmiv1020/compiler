%{
#include <stdio.h>

int vow_count = 0;
int cons_count = 0;
%}

%%
[aeiouAEIOU] { vow_count++; }
[a-zA-Z] { cons_count++; }
.|\n { /* ignore other characters */ }
%%

int yywrap()
{
    return 1;
}

int main()
{
    printf("Enter the string:\n");
    yylex();
    printf("Number of vowels: %d\n", vow_count);
    printf("Number of consonants: %d\n", cons_count);
    return 0;
}
